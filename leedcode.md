class Solution {

    int lengthOfLongestSubstring(string s) {
        unordered_map<char, int> hash;
        int ans = 0;
        int left = 0, right = 0;
        int n = s.size();
        while (right < n) {
            char c = s[right];
            if (hash.find(c) != hash.end() && hash[c] >= left) {
                left = hash[c] + 1;

            return false;
        }

            return true;

            if (i == 0) {
